# Components

AriaDocs provides React components for rendering MDX content on both server and client.

## `<MdxServer />`

Server component for rendering MDX. Use this when you want to render MDX directly in a server component.

### Usage

```tsx
import { MdxServer } from "@ariadocs/react";
// or
import { MdxServer } from "@ariadocs/react/server";

<MdxServer
  raw={mdxContent}
  options={{
    remarkPlugins: [],
    rehypePlugins: [],
  }}
  components={
    {
      // Custom components
    }
  }
/>;
```

### Props

| Property     | Type            | Required | Description            |
| ------------ | --------------- | -------- | ---------------------- |
| `raw`        | `string`        | Yes      | Raw MDX content string |
| `options`    | `DocsConfig`    | No       | Plugin configuration   |
| `components` | `MDXComponents` | No       | Custom MDX components  |

### Example

```tsx
import { MdxServer } from "@ariadocs/react";
import { remarkGfm, rehypePrism, rehypeSlug } from "@ariadocs/react/plugins";

export default function DocPage({ content }: { content: string }) {
  return (
    <MdxServer
      raw={content}
      options={{
        remarkPlugins: [remarkGfm],
        rehypePlugins: [rehypeSlug, rehypePrism],
      }}
      components={{
        h1: ({ children }) => (
          <h1 className="text-3xl font-bold">{children}</h1>
        ),
        pre: ({ children }) => (
          <pre className="bg-gray-900 p-4 rounded">{children}</pre>
        ),
      }}
    />
  );
}
```

## `<MdxClient />`

Client component for rendering serialized MDX. Use this when you need to render MDX in a client component.

### Usage

```tsx
"use client";
import { MdxClient } from "@ariadocs/react";
// or
import { MdxClient } from "@ariadocs/react/client";

<MdxClient serialized={serializedResult} mdxComponents={{}} />;
```

### Props

| Property        | Type            | Required | Description                      |
| --------------- | --------------- | -------- | -------------------------------- |
| `serialized`    | `string`        | Yes      | Serialized MDX from serializeMdx |
| `mdxComponents` | `MDXComponents` | No       | Custom MDX components            |

### Example

```tsx
// Server component - serialize the MDX
import { serializeMdx } from "@ariadocs/react";
import DocContent from "./DocContent";

export default async function Page() {
  const { serialized } = await serializeMdx({
    contentDir: "contents/docs",
    slug: "getting-started",
  });

  return <DocContent serialized={serialized} />;
}
```

```tsx
// Client component - render the MDX
"use client";
import { MdxClient } from "@ariadocs/react";

export default function DocContent({ serialized }: { serialized: string }) {
  return (
    <MdxClient
      serialized={serialized}
      mdxComponents={{
        h1: ({ children }) => (
          <h1 className="text-3xl font-bold">{children}</h1>
        ),
      }}
    />
  );
}
```

## Custom Components

Both `MdxServer` and `MdxClient` accept custom MDX components. This allows you to override default rendering.

### Available Elements

You can provide components for any HTML element or custom MDX component:

```tsx
const components = {
  // Headings
  h1: ({ children }) => <h1 className="text-4xl font-bold">{children}</h1>,
  h2: ({ children }) => <h2 className="text-3xl font-semibold">{children}</h2>,
  h3: ({ children }) => <h3 className="text-2xl">{children}</h3>,

  // Text
  p: ({ children }) => <p className="mb-4">{children}</p>,
  a: ({ href, children }) => (
    <a className="text-blue-600 hover:underline" href={href}>
      {children}
    </a>
  ),

  // Code
  pre: ({ children }) => (
    <pre className="bg-gray-900 p-4 rounded-lg">{children}</pre>
  ),
  code: ({ children }) => (
    <code className="bg-gray-100 px-1 rounded">{children}</code>
  ),

  // Lists
  ul: ({ children }) => <ul className="list-disc pl-6">{children}</ul>,
  ol: ({ children }) => <ol className="list-decimal pl-6">{children}</ol>,

  // Blockquote
  blockquote: ({ children }) => (
    <blockquote className="border-l-4 border-gray-300 pl-4 italic">
      {children}
    </blockquote>
  ),

  // Custom components
  Callout: ({ type, children }) => (
    <div className={`p-4 rounded callout-${type}`}>{children}</div>
  ),
};
```

### Using Custom Components

```tsx
import { MdxServer } from "@ariadocs/react";

const components = {
  Callout: ({
    type,
    children,
  }: {
    type: string;
    children: React.ReactNode;
  }) => (
    <div
      className={`p-4 rounded border-l-4 ${
        type === "warning"
          ? "border-yellow-500 bg-yellow-50"
          : "border-blue-500 bg-blue-50"
      }`}
    >
      {children}
    </div>
  ),
};

export default function Page({ content }: { content: string }) {
  return <MdxServer raw={content} components={components} />;
}
```

Then in your MDX:

```mdx
<Callout type="warning">This is a warning callout.</Callout>
```
